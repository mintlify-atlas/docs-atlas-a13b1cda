## min()

Gets the minimum item within the given bounds (the item with the smallest key).

### Signature

```typescript
async min(
  ctx: RunQueryCtx,
  ...opts: NamespacedOpts<{ bounds?: Bounds<K, ID> }, Namespace>
): Promise<Item<K, ID> | null>
```

### Parameters

<ParamField path="ctx" type="RunQueryCtx" required>
  Query context with `runQuery` method (e.g., from a Convex query or action)
</ParamField>

<ParamField path="opts" type="object">
  Optional configuration object
  
  <Expandable title="properties">
    <ParamField path="bounds" type="Bounds<K, ID>">
      Key range to search within. If not provided, searches all items.
    </ParamField>
    
    <ParamField path="namespace" type="Namespace">
      Namespace to query within (if using namespaced aggregates)
    </ParamField>
  </Expandable>
</ParamField>

### Returns

<ResponseField name="item" type="Item<K, ID> | null">
  The item with the smallest key, or `null` if no items exist within the bounds
  
  <Expandable title="Item properties">
    <ResponseField name="key" type="K">
      The sort key of the item
    </ResponseField>
    
    <ResponseField name="id" type="ID">
      The unique identifier of the item
    </ResponseField>
    
    <ResponseField name="sumValue" type="number">
      The sum value associated with this item
    </ResponseField>
  </Expandable>
</ResponseField>

### Example

```typescript
export const lowestScore = query({
  handler: async (ctx) => {
    const item = await aggregateByScore.min(ctx);
    if (!item) return null;
    return await ctx.db.get(item.id);
  },
});
```

## random()

Gets a uniformly random item within the given bounds.

### Signature

```typescript
async random(
  ctx: RunQueryCtx,
  ...opts: NamespacedOpts<{ bounds?: Bounds<K, ID> }, Namespace>
): Promise<Item<K, ID> | null>
```

### Parameters

<ParamField path="ctx" type="RunQueryCtx" required>
  Query context with `runQuery` method
</ParamField>

<ParamField path="opts" type="object">
  Optional configuration object
  
  <Expandable title="properties">
    <ParamField path="bounds" type="Bounds<K, ID>">
      Key range to search within. If not provided, selects from all items.
    </ParamField>
    
    <ParamField path="namespace" type="Namespace">
      Namespace to query within (if using namespaced aggregates)
    </ParamField>
  </Expandable>
</ParamField>

### Returns

<ResponseField name="item" type="Item<K, ID> | null">
  A randomly selected item, or `null` if no items exist within the bounds
</ResponseField>

### Example

```typescript
export const randomScore = query({
  handler: async (ctx) => {
    const item = await aggregateByScore.random(ctx);
    if (!item) return null;
    return await ctx.db.get(item.id);
  },
});
```

### Implementation Note

The `random()` method works by:
1. Counting all items within the bounds using `count()`
2. Generating a random index between 0 and count-1
3. Retrieving the item at that index using `at()`

```typescript
async random(
  ctx: RunQueryCtx,
  ...opts: NamespacedOpts<{ bounds?: Bounds<K, ID> }, Namespace>
): Promise<Item<K, ID> | null> {
  const count = await this.count(ctx, ...opts);
  if (count === 0) {
    return null;
  }
  const index = Math.floor(Math.random() * count);
  return await this.at(ctx, index, ...opts);
}
```
