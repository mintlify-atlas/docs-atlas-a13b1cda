## sum()

Adds up the `sumValue` of items between the given bounds.

### Signature

```typescript
async sum(
  ctx: RunQueryCtx,
  ...opts: NamespacedOpts<{ bounds?: Bounds<K, ID> }, Namespace>
): Promise<number>
```

### Parameters

<ParamField path="ctx" type="RunQueryCtx" required>
  Query context with `runQuery` method (e.g., from a Convex query or action)
</ParamField>

<ParamField path="opts" type="object">
  Optional configuration object
  
  <Expandable title="properties">
    <ParamField path="bounds" type="Bounds<K, ID>">
      Key range to sum items within. If not provided, sums all items.
      
      <Expandable title="bounds properties">
        <ParamField path="lower" type="Bound<K, ID>">
          Lower bound with `key`, optional `id`, and `inclusive` flag
        </ParamField>
        
        <ParamField path="upper" type="Bound<K, ID>">
          Upper bound with `key`, optional `id`, and `inclusive` flag
        </ParamField>
        
        <ParamField path="prefix" type="K">
          Prefix to match (alternative to specifying lower/upper bounds)
        </ParamField>
      </Expandable>
    </ParamField>
    
    <ParamField path="namespace" type="Namespace">
      Namespace to query within (if using namespaced aggregates)
    </ParamField>
  </Expandable>
</ParamField>

### Returns

<ResponseField name="sum" type="number">
  The sum of `sumValue` for all items within the specified bounds. If an item has no `sumValue`, it's treated as zero.
</ResponseField>

### Example

<Tabs>
  <Tab title="Sum All Values">
    ```typescript
    export const sumNumbers = query({
      handler: async (ctx) => {
        return await aggregateScoreByUser.sum(ctx);
      },
    });
    ```
  </Tab>
  
  <Tab title="User Average Score">
    ```typescript
    export const userAverageScore = query({
      args: {
        name: v.string(),
      },
      handler: async (ctx, args) => {
        const count = await aggregateScoreByUser.count(ctx, {
          bounds: { prefix: [args.name] },
        });
        if (!count) return null;
        
        const sum = await aggregateScoreByUser.sum(ctx, {
          bounds: { prefix: [args.name] },
        });
        return sum / count;
      },
    });
    ```
  </Tab>
</Tabs>

## sumBatch()

Batch version of `sum()` - sums items for multiple bounds in a single call.

### Signature

```typescript
async sumBatch(
  ctx: RunQueryCtx,
  queries: NamespacedOptsBatch<{ bounds?: Bounds<K, ID> }, Namespace>
): Promise<number[]>
```

### Parameters

<ParamField path="ctx" type="RunQueryCtx" required>
  Query context with `runQuery` method
</ParamField>

<ParamField path="queries" type="Array<object>" required>
  Array of query objects, each with optional bounds and namespace
  
  <Expandable title="query object properties">
    <ParamField path="bounds" type="Bounds<K, ID>">
      Key range to sum items within
    </ParamField>
    
    <ParamField path="namespace" type="Namespace">
      Namespace to query within
    </ParamField>
  </Expandable>
</ParamField>

### Returns

<ResponseField name="sums" type="number[]">
  Array of sums, one for each query in the input array
</ResponseField>

### Example

```typescript
export const totalScoresByUsers = query({
  args: { users: v.array(v.string()) },
  handler: async (ctx, { users }) => {
    const queries = users.map(name => ({
      bounds: { prefix: [name] },
    }));
    
    const sums = await aggregateScoreByUser.sumBatch(ctx, queries);
    
    return users.map((name, i) => ({
      name,
      totalScore: sums[i],
    }));
  },
});
```

### Use Case

The batch methods are optimized for querying multiple ranges efficiently. Instead of making multiple separate calls, `sumBatch()` performs all queries in a single operation, reducing latency and improving performance.
